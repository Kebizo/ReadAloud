function updateOriginalWordCount() { const text = originalTextArea.value.trim(); let wordCount = 0; let characterCount = 0; const wordRegex = /\b\w+\b/g; const cjkRegex = /[\u3040-\u30FF\u3400-\u4DBF\u4E00-\u9FFF\uF900-\uFAFF]/g; const words = text.match(wordRegex); if (words) { wordCount = words.length; } const cjkCharacters = text.match(cjkRegex); if (cjkCharacters) { characterCount = cjkCharacters.length; } const totalCount = wordCount + characterCount; originalWordCountDiv.textContent = `Word/Character Count: ${totalCount}`; originalTextContent = text; }
