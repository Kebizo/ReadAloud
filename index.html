            padding: 10px 20px;

        #transcribedText {

            margin-top: 20px;

            font-size: 18px;

<body>

    <h1>Read Aloud Practice</h1>

    <textarea id="originalText" placeholder="Paste your text here"></textarea><br><br>

    <button id="startButton">Start Reading</button>

    <div id="transcribedText"></div>


    <script>

        // Check for browser support

        window.SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;


        if ('SpeechRecognition' in window) {

            const recognition = new SpeechRecognition();

            recognition.lang = 'en-US';

            recognition.interimResults = false;

            recognition.continuous = false;


            const startButton = document.getElementById('startButton');

            const originalTextArea = document.getElementById('originalText');

            const transcribedDiv = document.getElementById('transcribedText');


            startButton.addEventListener('click', () => {

                if (originalTextArea.value.trim() === '') {

                    alert('Please paste text into the input area before starting.');

                    return;

                }

                recognition.start();

            });


            recognition.addEventListener('result', (event) => {

                const transcript = event.results[0][0].transcript;

                compareTexts(originalTextArea.value, transcript);

            });


            recognition.addEventListener('end', () => {

                recognition.stop();

            });


            function compareTexts(original, transcribed) {

                const originalWords = original.trim().split(/\s+/);

                const transcribedWords = transcribed.trim().split(/\s+/);


                let resultHTML = '';

                for (let i = 0; i < originalWords.length; i++) {

                    const originalWord = originalWords[i] || '';

                    const transcribedWord = transcribedWords[i] || '';


                    if (originalWord.toLowerCase() === transcribedWord.toLowerCase()) {

                        resultHTML += `<span class="correct">${transcribedWord} </span>`;

                    } else {

                        resultHTML += `<span class="mistake">${transcribedWord} </span>`;

                    }

                }

                transcribedDiv.innerHTML = `<p><strong>Your Transcription:</strong></p>${resultHTML}`;

            }

        } else {

            alert('Sorry, your browser does not support speech recognition.');

        }

    </script>

</body>
